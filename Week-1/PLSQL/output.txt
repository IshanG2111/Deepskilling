SQL> CREATE OR REPLACE PROCEDURE TransferFunds(
  2      p_source NUMBER,
  3      p_target NUMBER,
  4      p_amount NUMBER
  5  )
  6  IS
  7      v_balance NUMBER;
  8  BEGIN
  9  
 10      SELECT Balance
 11      INTO v_balance
 12      FROM Accounts
 13      WHERE AccountID = p_source;
 14  
 15      IF v_balance >= p_amount THEN
 16  
 17          UPDATE Accounts
 18          SET Balance = Balance - p_amount
 19          WHERE AccountID = p_source;
 20  
 21          UPDATE Accounts
 22          SET Balance = Balance + p_amount
 23          WHERE AccountID = p_target;
 24  
 25          COMMIT;

 27          DBMS_OUTPUT.PUT_LINE('Transfer Successful');

 29      ELSE

 35  END;
